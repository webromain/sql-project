```sql
CREATE TABLE utilisateur (
    id_user INT AUTO_INCREMENT PRIMARY KEY,
    nom VARCHAR(100) NOT NULL,
    prenom VARCHAR(100) NOT NULL,
    email VARCHAR(255) NOT NULL UNIQUE,
    mot_de_passe VARCHAR(255) NOT NULL,
    role VARCHAR(20) NOT NULL DEFAULT 'participant',
    date_inscription DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE evenement (
    id_event INT AUTO_INCREMENT PRIMARY KEY,
    titre VARCHAR(255) NOT NULL,
    description TEXT,
    date_event DATETIME NOT NULL,
    lieu VARCHAR(255) NOT NULL,
    capacite_max INT NOT NULL,
    prix DECIMAL(10,2) NOT NULL DEFAULT 0.00
);

CREATE TABLE speaker (
    id_speaker INT AUTO_INCREMENT PRIMARY KEY,
    id_user INT NOT NULL UNIQUE,
    bio TEXT,

    CONSTRAINT fk_speaker_user
    FOREIGN KEY (id_user)
    REFERENCES utilisateur(id_user)
    ON DELETE CASCADE
    ON UPDATE CASCADE
);

CREATE TABLE session (
    id_session INT AUTO_INCREMENT PRIMARY KEY,
    id_event INT NOT NULL,
    id_speaker INT NOT NULL,
    titre VARCHAR(255) NOT NULL,
    heure_debut DATETIME NOT NULL,
    heure_fin DATETIME NOT NULL,

    CONSTRAINT fk_session_event
    FOREIGN KEY (id_event)
    REFERENCES evenement(id_event)
    ON DELETE CASCADE
    ON UPDATE CASCADE,

    CONSTRAINT fk_session_speaker
    FOREIGN KEY (id_speaker)
    REFERENCES speaker(id_speaker)
    ON DELETE CASCADE
    ON UPDATE CASCADE
);

CREATE TABLE inscription (
    id_registration INT AUTO_INCREMENT PRIMARY KEY,
    id_user INT NOT NULL,
    id_event INT NOT NULL,
    date_inscription DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    statut_paiement VARCHAR(20) NOT NULL DEFAULT 'en_attente',

    CONSTRAINT fk_inscription_user
    FOREIGN KEY (id_user)
    REFERENCES utilisateur(id_user)
    ON DELETE CASCADE
    ON UPDATE CASCADE,

    CONSTRAINT fk_inscription_event
    FOREIGN KEY (id_event)
    REFERENCES evenement(id_event)
    ON DELETE CASCADE
    ON UPDATE CASCADE,

    CONSTRAINT unique_inscription
    UNIQUE (id_user, id_event)
);

CREATE TABLE avis (
    id_review INT AUTO_INCREMENT PRIMARY KEY,
    id_user INT NOT NULL,
    id_event INT NOT NULL,
    note INT NOT NULL,
    commentaire TEXT,
    date_review DATETIME DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_avis_user
    FOREIGN KEY (id_user)
    REFERENCES utilisateur(id_user)
    ON DELETE CASCADE
    ON UPDATE CASCADE,

    CONSTRAINT fk_avis_event
    FOREIGN KEY (id_event)
    REFERENCES evenement(id_event)
    ON DELETE CASCADE
    ON UPDATE CASCADE,

    CONSTRAINT unique_avis
    UNIQUE (id_user, id_event)
);

```
